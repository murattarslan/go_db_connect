# go_db_connect
connect db with go

dao paketinin içerisinde conf.go dosyasında db için konfigürasyon bilgileri bulunacak

örnek

<---------------------------->

package dao

const (
	host     = "localhost"
	port     = "5432"
	user     = "user"
	password = "password"
	dbname   = "dbname"
)
